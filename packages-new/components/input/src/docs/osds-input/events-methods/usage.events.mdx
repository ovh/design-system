For more details go to [Events](#events)

#### OdsValueChange

```html
<osds-input id="odsInput-1" type="number" placeholder="Placeholder" min="1" max="4" step="2" color="primary"></osds-input>

<script>
  const odsInput1 = document.querySelector('osds-input#odsInput-1');

  odsInput1.addEventListener('odsValueChange', (event) => {
    console.log("odsValueChange event", event.detail);

    odsInput1.getValidity().then((validity) => {
      console.log("getValidity odsInput-1 from Event", validity);
    });
  })
</script>

```
