### With a trigger

export const buttonTrigger = () => {
    setTimeout(() => {
        const elemCollapsibleUsage = document.getElementById('collapsibleUsage');
        elemCollapsibleUsage?.addEventListener('odsCollapsibleToggle', event => {
            console.log('(html) addEventListener odsCollapsibleToggle: ', event.detail);
        });
        const buttonTriggerCollapsibleUsage = document.getElementById('buttonTriggerCollapsibleUsage');
        buttonTriggerCollapsibleUsage?.addEventListener('click', () => {
            elemCollapsibleUsage.opened = !elemCollapsibleUsage.opened;
        });
    }, 0);
};
