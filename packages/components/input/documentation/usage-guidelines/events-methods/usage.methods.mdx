#### getValidity

```html
<osds-input type="number" placeholder="Placeholder" min="1" max="4" step="2" color="primary"></osds-input>

<script>
  const odsInput = document.querySelector('osds-input');

  async function componentOnLoad() {
    await customElements.whenDefined('osds-input');
    odsInput.getValidity().then((validity) => {
      console.log("getValidity odsInput-1", validity);
    });
  }

  componentOnLoad();
</script>
```
