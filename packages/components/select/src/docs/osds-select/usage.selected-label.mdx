export const selectedLabel = () => {
    setTimeout(() => {
        const selectWithSelectedlabel = document.getElementById('withSelectedlabel');
        const slotSelectedLabel = selectWithSelectedlabel?.querySelector('[slot="selectedLabel"]')
        selectWithSelectedlabel?.addEventListener('odsValueChange', (event) => {
            // remove all previous child
            while (slotSelectedLabel?.firstChild) {
            slotSelectedLabel?.removeChild(slotSelectedLabel?.firstChild);
            }
            const span = document.createElement('span');
            span.innerText = event.detail.value;
            const icon = document.createElement('osds-icon')
            icon.name = 'home';
            icon.size = 'sm';
            slotSelectedLabel?.appendChild(icon);
            slotSelectedLabel?.appendChild(span);
        });
    }, 0);
};
