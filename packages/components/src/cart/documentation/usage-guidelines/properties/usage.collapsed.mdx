Hide header content while `collapsed`:

export const displayHeader = () => {
    setTimeout(() => {
        const cart = document.querySelector('osds-cart');
        const cartHeader = document.querySelector('#cart-header');
        cart.setAttribute('collapsed', true);
        cart.collapsed ? cartHeader.innerText = '' : cartHeader.innerText = 'Header';
    }, 0);
};
