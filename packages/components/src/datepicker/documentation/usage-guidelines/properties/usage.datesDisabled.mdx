export const displayDatesDisabledDatepicker = () => {
    setTimeout(() => {
        const datepickerDatesDisabledUsage = document.querySelector('#datepicker-dates-disabled');
        datepickerDatesDisabledUsage.datesDisabled = [new Date()];
    }, 0);
};

<osds-datepicker id='datepicker-dates-disabled' inline></osds-datepicker>

<>{ displayDatesDisabledDatepicker() }</>
