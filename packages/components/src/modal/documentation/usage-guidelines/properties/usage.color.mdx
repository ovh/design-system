#### Info (default)
export const displayInfoModal = () => {
  setTimeout(() => {
    const toggleModal = document.getElementById('toggle-modal-2');
    const modal = document.getElementById('modal-info');
    toggleModal.addEventListener('click', () => {
      modal.open();
    });
  }, 0);
};

<osds-button id="toggle-modal-2" color='primary' inline>Toggle Modal in info</osds-button>

<osds-modal id="modal-info" masked color='info'>Modal</osds-modal>

<>{ displayInfoModal() }</>

```html
<osds-modal color='info'>Modal</osds-modal>
```
