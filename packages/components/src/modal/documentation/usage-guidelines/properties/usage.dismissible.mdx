A Modal can be undismissible:

export const displayDismissibleModal = () => {
  setTimeout(() => {
    const toggleModal = document.getElementById('toggle-modal-6');
    const modal = document.getElementById('modal-dismissible');
    toggleModal.addEventListener('click', () => {
      modal.open();
    });
    const closeBtn = document.getElementById('close-button');
    closeBtn.addEventListener('click', () => {
          modal.close();
        });
  }, 0);
};

<osds-button id="toggle-modal-6" color='primary' inline>Toggle dismissible modal</osds-button>

<osds-modal id="modal-dismissible" masked dismissible='false'>
  Modal
  <osds-button slot='actions' color='primary' variant='stroked' id='close-button'>Close modal</osds-button>
</osds-modal>
