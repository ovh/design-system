## Usage

### Default
export const displayPhoneNumber = () => {
    setTimeout(() => {
        const phone = document.querySelector('#phone');
        if (phone) {
          phone.addEventListener('keydown', (e) => {
            e.stopPropagation();
          });
      }
    }, 0);
};
