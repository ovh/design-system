### Options (with Select)

export const fillOptions = () => {
    setTimeout(() => {
        const searchbarWithOptions = document.getElementById('searchbarWithOptions');
        searchbarWithOptions.options = [{ label: 'option1', value: '1' }, { label: 'option2', value: '2' }, { label: 'option3', value: '3' }]
    }, 0);
};

<osds-search-bar id="searchbarWithOptions"></osds-search-bar>

 ]">
