You can indicate which panel you want to select with the `panel` attribute.

<osds-tabs panel ='tab3'>
  <osds-tab-bar slot='top'>
    <osds-tab-bar-item panel='tab1'>Tab 1</osds-tab-bar-item>
    <osds-tab-bar-item panel='tab2'>Tab 2</osds-tab-bar-item>
    <osds-tab-bar-item panel='tab3'>Tab 3</osds-tab-bar-item>
  </osds-tab-bar>
  <osds-tab-panel name='tab1'>Content for Tab 1</osds-tab-panel>
  <osds-tab-panel name='tab2'>Content for Tab 2</osds-tab-panel>
  <osds-tab-panel name='tab3'>Content for Tab 3</osds-tab-panel>
</osds-tabs>
