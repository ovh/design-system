For more details go to [Events](#events)

#### OdsValueChange

```html
<osds-textarea id="odsTextArea-1" placeholder="Placeholder"></osds-textarea>

<script>
  const odsTextArea1 = document.querySelector('osds-textarea#odsTextArea-1');

  odsTextArea1.addEventListener('odsValueChange', (event) => {
    console.log("odsValueChange event", event.detail);

    odsTextArea1.getValidity().then((validity) => {
      console.log("getValidity odsTextArea-1 from Event", validity);
    });
  })
</script>

```
