#### getValidity

```html
<osds-textarea placeholder="Placeholder"></osds-textarea>

<script>
  const odsTextArea = document.querySelector('osds-textarea');

  async function componentOnLoad() {
    await customElements.whenDefined('osds-textarea');
    odsTextArea.getValidity().then((validity) => {
      console.log("getValidity odsTextArea-1", validity);
    });
  }

  componentOnLoad();
</script>
```
