# formatPrice

## Examples


### Currencies

```tsx
{
  globals: {
    imports: `import { formatPrice } from '@ovhcloud/ods-react';`
  },
  parameters: {
    docs: {
      source: {
        ...staticSourceRenderConfig()
      }
    }
  },
  tags: ['!dev'],
  render: ({}) => <>
      <p>Euro: {formatPrice(123456.789, 'fr-FR', 'EUR')}</p>
      <p>Dollar: {formatPrice(123456.789, 'fr-FR', 'USD')}</p>
      <p>Yen: {formatPrice(123456.789, 'fr-FR', 'JPY')}</p>
    </>
}
```
