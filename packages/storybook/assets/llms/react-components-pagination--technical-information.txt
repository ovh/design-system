## Examples

---

### Default

```jsx
<Pagination totalItems={5000}>
  <PaginationPages />
</Pagination>
```

### Controlled

```jsx
const [page, setPage] = useState(1);
  function handlePageChange({
    page  }: PaginationPageChangeDetail) {
    setPage(page);
  }
  return <Pagination onPageChange={handlePageChange} page={page} totalItems={500}>
      <PaginationPages />
    </Pagination>;
}
```

### Disabled

```jsx
<Pagination
  disabled
  totalItems={500}
>
  <PaginationPages />
</Pagination>
```

### Items per page

```jsx
<Pagination
  pageSize={25}
  totalItems={500}
>
  <PaginationPages />
</Pagination>
```

### Sibling count

```jsx
<Pagination
  defaultPage={5}
  siblingCount={2}
  totalItems={500}
>
  <PaginationPages />
</Pagination>
```

### With tooltip labels

```jsx
<Pagination
  labelTooltipNext="Go to next page"
  labelTooltipPrev="Go to previous page"
  totalItems={500}
>
  <PaginationPages />
</Pagination>
```
