```jsx
    <QuantityInput />
  </QuantityControl>
</Quantity>
```

### Disabled

```jsx

  <QuantityControl>
    <QuantityInput />
  </QuantityControl>
</Quantity>
```

### Readonly

```jsx
<Quantity readOnly>
  <QuantityControl>
    <QuantityInput />
  </QuantityControl>
</Quantity>
```

### Max

```jsx
<Quantity max={10}>
  <QuantityControl>
    <QuantityInput />
  </QuantityControl>
</Quantity>
```

### Min

```jsx
<Quantity min={0}>
  <QuantityControl>
    <QuantityInput />
  </QuantityControl>
</Quantity>
```

### Step

```jsx

  <QuantityControl>
    <QuantityInput />
  </QuantityControl>
</Quantity>
```

### Form field

Set a quantity:

```jsx
<FormField>
  <FormFieldLabel>
    Set a quantity:  </FormFieldLabel>
  <Quantity>
    <QuantityControl>
      <QuantityInput />
    </QuantityControl>
  </Quantity>
</FormField>
```
