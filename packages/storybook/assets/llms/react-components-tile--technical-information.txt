## Examples

---

### Default

This is a tile content.

```jsx
<Tile>
  This is a tile content.
</Tile>
```

### Disabled

This tile is disabled.

```jsx
<Tile disabled>
  This tile is disabled.
</Tile>
```

### Selected

This tile is selected.

```jsx
<Tile selected>
  This tile is selected.
</Tile>
```

### With Checkbox

Checkbox tile

```jsx
const [selected, setSelected] = useState(false);
  return <Tile selected={selected}>
      <Checkbox checked={selected} onCheckedChange={detail => setSelected(detail.checked === true)}>
        <div style={{
        display: 'flex',
        alignItems: 'center',
        gap: '8px',
        padding: '16px'
      }}>
          <CheckboxControl />
          <CheckboxLabel>Checkbox tile</CheckboxLabel>
        </div>
      </Checkbox>
    </Tile>;
}
```

### With RadioGroup

Radio 1

Radio 2

```jsx
const [selected, setSelected] = useState('radio-1');
  return <RadioGroup value={selected} onValueChange={detail => setSelected(detail.value || '')}>
      <div>
        <Tile selected={selected === 'radio-1'}>
          <Radio value={'radio-1'}>
            <div style={{
            display: 'flex',
            alignItems: 'center',
            gap: '8px',
            padding: '16px'
          }}>
              <RadioControl />
              <RadioLabel>
                Radio 1              </RadioLabel>
            </div>
          </Radio>
        </Tile>
      </div>
      <div>
        <Tile selected={selected === 'radio-2'}>
          <Radio value={'radio-2'}>
            <div style={{
            display: 'flex',
            alignItems: 'center',
            gap: '8px',
            padding: '16px'
          }}>
              <RadioControl />
              <RadioLabel>
                Radio 2              </RadioLabel>
            </div>
          </Radio>
        </Tile>
      </div>
    </RadioGroup>;
}
```
