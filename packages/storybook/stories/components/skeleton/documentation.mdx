<BestPractices
  donts={[
    '- Avoid using a Skeleton for static content',
    '- Use for interactive elements',
  ]}
  dos={[
    '- Prefer using a Skeleton for dynamic content except it is necessary for your use case',
    '- Use a Skeleton if content loading is going to take a few seconds',
    '- Use when the page or a part of the page is loading or updating',
  ]}
/>
